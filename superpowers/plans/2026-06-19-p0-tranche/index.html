        self.assertTrue(issubclass(XPConnectionError, XPWebAPIError))

    def test_beacon_error_is_connection_error(self):
        self.assertTrue(issubclass(XPBeaconError, XPConnectionError))

    def test_timeout_error_is_xpwebapi_error(self):
        self.assertTrue(issubclass(XPTimeoutError, XPWebAPIError))

    def test_version_error_is_xpwebapi_error(self):
        self.assertTrue(issubclass(XPVersionError, XPWebAPIError))

    def test_context_kwargs(self):
        err = XPWebAPIError("boom", host="127.0.0.1", port=8086)
        self.assertEqual(str(err), "boom")
        self.assertEqual(err.context, {"host": "127.0.0.1", "port": 8086})

    def test_context_empty_by_default(self):
        err = XPWebAPIError("oops")
        self.assertEqual(err.context, {})

    def test_beacon_error_context(self):
        err = XPBeaconError("no beacon", timeout=3.0)
        self.assertEqual(err.context, {"timeout": 3.0})
        self.assertIsInstance(err, XPConnectionError)

    def test_timeout_error_context(self):
        err = XPTimeoutError("timed out", host="10.0.0.1")
        self.assertEqual(err.context, {"host": "10.0.0.1"})

    def test_version_error_context(self):
        err = XPVersionError("unsupported", version="10.40")
        self.assertEqual(err.context, {"version": "10.40"})

        for cls in (XPConnectionError, XPBeaconError, XPTimeoutError, XPVersionError):

class TestDatarefParseRawValue(unittest.TestCase):
    def setUp(self):
        self.api = MagicMock(spec=API)
        self.api.use_cache = False
        self.api.get_rest_meta = MagicMock()

    def _make_dataref(self, path, value_type="int"):
        meta = DatarefMeta(name=path, value_type=value_type, is_writable=False, id=1)
        self.api.get_rest_meta.return_value = meta
        d = Dataref(path=path, api=self.api)
        return d

    def test_parse_scalar_int(self):
        d = self._make_dataref("sim/test/val", "int")
        self.assertEqual(d.parse_raw_value(42), 42)

    def test_parse_scalar_float(self):
        d = self._make_dataref("sim/test/val", "float")
        self.assertAlmostEqual(d.parse_raw_value(3.14), 3.14)

    def test_parse_array_whole(self):
        d = self._make_dataref("sim/test/arr", "int_array")
        self.assertEqual(d.parse_raw_value([1, 2, 3]), [1, 2, 3])

    def test_parse_array_single_index(self):
        meta = DatarefMeta(name="sim/test/arr", value_type="int_array", is_writable=False, id=1)
        meta.append_index(2)
        self.api.get_rest_meta.return_value = meta
        d = Dataref(path="sim/test/arr[2]", api=self.api)
        self.assertEqual(d.parse_raw_value([10, 20, 30]), 20)

    def test_parse_base64_data(self):
        d = self._make_dataref("sim/test/data", "data")
        import base64
        encoded = base64.b64encode(b"hello").decode("ascii")
        result = d.parse_raw_value(encoded)
        self.assertEqual(result, b"hello")

    def test_parse_invalid_meta(self):
        self.api.get_rest_meta.return_value = None
        d = Dataref(path="sim/test/val", api=self.api)
        self.assertIsNone(d.parse_raw_value(42))

def make_beacon_packet(host="127.0.0.1", port=49000, hostname="testhost", xplane_version=121400, role=1):
    header = b"BECN\x00"
    major = 1
    minor = 2
    app_id = 1
    data = struct.pack("<BBiiIH", major, minor, app_id, xplane_version, role, port)
    hostname_bytes = hostname.encode() + b"\x00"
    return header + data + hostname_bytes

def make_rref_packet(datarefs_and_values):
    header = b"RREF,"
    packet = header
    for idx, value in datarefs_and_values:
        packet += struct.pack("<if", idx, value)
    return packet
